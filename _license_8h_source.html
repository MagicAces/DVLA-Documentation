28 // Helper function to break strings into segments depending a given delimiter(e.g " ", "-")
29 vector<string> tokenizeString(string word, char* delimiter) {
30 vector<string> words;
31 char wordArray [word.length() + 1] ;
32
33 for(int i = 0; i < word.length() + 1; i++)
34 wordArray[i] = word[i];
35
36 char *subword = strtok(wordArray, delimiter);
37
38 while(subword != NULL) {
39 words.push_back(subword);
40 subword = strtok(NULL, delimiter);
41 }
42
43 return words; // Returns a vector storing all the segments
44 }

227 int registryYear = stoi(parts.at(0));

234 int currentMonth = 1 + ltm->tm_mon;
235
236 // Checks to see if the difference between the current date and the registration date is at least 2
237 if(currentYear - registryYear >= 2 && currentMonth - registryMonth >= 0)
238 return true;
